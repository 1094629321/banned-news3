  <ok href="https://www.ntdtv.com/gb/雕像.htm">

   川普
  </ok>
  <ok href="https://www.ntdtv.com/gb/雕像.htm">
   雕像
  </ok>

  <ok href="https://www.ntdtv.com/gb/淘宝.htm">
   淘宝
  </ok>
